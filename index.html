// Infinite Loop Slider Control Logic const track = document.getElementById('sliderTrack'); const slides = document.querySelectorAll('.carousel-slide'); let currentSlide = 1; // Starts at 1 because index 0 is the cloned last slide const totalRealSlides = 4; let isMoving = false; // Safe click guard flag // Initialize position to look at first real slide track.style.transform = `translateX(-${currentSlide * 100}%)`; function moveSlider(direction) { // If an animation is already running, ignore incoming clicks if (isMoving) return; isMoving = true; currentSlide += direction; track.style.transition = "transform 0.5s cubic-bezier(0.25, 1, 0.5, 1)"; track.style.transform = `translateX(-${currentSlide * 100}%)`; } // Listen for the CSS transition to fully finish before handling loop resets track.addEventListener('transitionend', () => { isMoving = false; // Release the click lock if (currentSlide === 0) { // Instantly teleport to the true end slide counterpart track.style.transition = "none"; currentSlide = totalRealSlides; track.style.transform = `translateX(-${currentSlide * 100}%)`; } else if (currentSlide === totalRealSlides + 1) { // Instantly teleport to the true beginning slide counterpart track.style.transition = "none"; currentSlide = 1; track.style.transform = `translateX(-${currentSlide * 100}%)`; } }); // System to swap between global page layouts smoothly function switchPage(pageTarget) { document.getElementById('nav-home').classList.remove('active'); document.getElementById('nav-portfolio').classList.remove('active'); document.getElementById('page-home').classList.remove('active-section'); document.getElementById('page-portfolio').classList.remove('active-section'); if(pageTarget === 'home') { document.getElementById('nav-home').classList.add('active'); document.getElementById('page-home').classList.add('active-section'); } else if (pageTarget === 'portfolio') { document.getElementById('nav-portfolio').classList.add('active'); document.getElementById('page-portfolio').classList.add('active-section'); } window.scrollTo(0, 0); } // Subtab management inside the Portfolio view function switchPortfolioTab(type) { document.getElementById('tab-long').classList.remove('active'); document.getElementById('tab-short').classList.remove('active'); document.getElementById('subgrid-longform').classList.remove('active-subgrid'); document.getElementById('subgrid-shortform').classList.remove('active-subgrid'); if(type === 'longform') { document.getElementById('tab-long').classList.add('active'); document.getElementById('subgrid-longform').classList.add('active-subgrid'); } else if(type === 'shortform') { document.getElementById('tab-short').classList.add('active'); document.getElementById('subgrid-shortform').classList.add('active-subgrid'); } }
